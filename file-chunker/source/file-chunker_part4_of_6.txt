 function splitText(text, maxBytes, mode) {
 const target = rawTarget(maxBytes);
 const out = [];
 const len = text.length;
 if (len === 0) return [''];
 let pos = 0;
 while (pos < len) {
 let end = Math.min(pos + target, len);
 if (end < len && mode !== 'hard') {
 const slice = text.slice(pos, end);
 let breakAt = -1;
 if (mode === 'smart') { breakAt = slice.lastIndexOf('\n\n'); if (breakAt !== -1) breakAt += 2; }
 if (breakAt === -1) { const nl = slice.lastIndexOf('\n'); if (nl !== -1) breakAt = nl + 1; }
 if (breakAt > target * 0.5) end = pos + breakAt;
 }
 out.push(text.slice(pos, end));
 pos = end;
 }
 return out;
 }
